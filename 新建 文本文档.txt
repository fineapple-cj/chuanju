  <script>
    // 悬浮窗拖动功能
    const floatBox = document.getElementById('lingduo-float');
    let isDragging = false;
    let startX, startY, offsetX, offsetY;

    floatBox.addEventListener('mousedown', (e) => {
      isDragging = true;
      startX = e.clientX;
      startY = e.clientY;
      offsetX = floatBox.offsetLeft;
      offsetY = floatBox.offsetTop;
      floatBox.style.cursor = 'grabbing';
    });

    document.addEventListener('mousemove', (e) => {
      if (!isDragging) return;
      const dx = e.clientX - startX;
      const dy = e.clientY - startY;
      floatBox.style.left = `${offsetX + dx}px`;
      floatBox.style.top = `${offsetY + dy}px`;
    });

    document.addEventListener('mouseup', () => {
      isDragging = false;
      floatBox.style.cursor = 'move';
    });

    // 核心交互逻辑
    const avatar = document.getElementById('avatar');
    const chatBubble = document.getElementById('chat-bubble');
    const talkBtn = document.getElementById('talk-btn');
    const closedMouth = ""https://ld-1407373774.cos.ap-chengdu.myqcloud.com/ld/%E4%BC%B6%E6%9C%B5.png"";
    const openMouth = ""https://ld-1407373774.cos.ap-chengdu.myqcloud.com/ld/%E4%BC%B6%E6%9C%B5.png"";

    // 语音识别+川剧AI回复
    talkBtn.addEventListener('click', startChat);
    function startChat() {
      const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
      if (!SpeechRecognition) {
        showChat('你的浏览器不支持语音功能，建议用Chrome哦~');
        return;
      }

      const rec = new SpeechRecognition();
      rec.lang = 'zh-CN';
      showChat('伶朵在听呀，说说你想了解川剧什么？');
      
      rec.onresult = (e) => {
        const userText = e.results[0][0].transcript;
        showChat(`你：${userText}`);
        getAIReply(userText);
      };
      rec.start();
    }

    // 调用川剧专属AI（替换为你的豆包API）
    async function getAIReply(userText) {
      try {
        const response = await fetch('https://your-doubao-api.com/chat', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            message: userText,
            system_prompt: "你是川剧元视界的伶朵，可爱的科技绵羊川剧旦角，用四川方言+俏皮语气，介绍川剧脸谱、唱腔、绝活，回复不超过3句话"
          })
        });
        const data = await response.json();
        const reply = data.reply || "伶朵带你耍川剧！变脸有3种手法，红脸代表忠义，要不要听听高腔唱段呀？";
        showChat(`伶朵：${reply}`);
        speak(reply);
      } catch (error) {
        const fallbackReply = "伶朵来科普！川剧有昆、高、胡、弹、灯五种声腔，变脸超神奇~";
        showChat(`伶朵：${fallbackReply}`);
        speak(fallbackReply);
      }
    }

    // 语音合成+口型同步
    function speak(text) {
      const utterance = new SpeechSynthesisUtterance(text);
      utterance.lang = 'zh-CN';
      utterance.rate = 0.9; // 放慢语速更贴合可爱人设
      avatar.src = openMouth;
      utterance.onend = () => { avatar.src = closedMouth; };
      speechSynthesis.speak(utterance);
    }

    // 聊天气泡显示
    function showChat(text) {
      chatBubble.textContent = text;
      chatBubble.style.display = 'block';
      setTimeout(() => { chatBubble.style.display = 'none'; }, 8000);
    }
  </script>
